#include <stdio.h>

// Función para verificar si un número es primo
int esPrimo(int num) {
    if (num <= 1) {
        return 0;  // No es primo
    }

    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return 0;  // No es primo
        }
    }

    return 1;  // Es primo
}

int main() {
    int numero;

    // Entrada del usuario
    printf("Ingresa un numero de cuatro cifras: ");
    scanf("%d", &numero);

    // Verificar si el número tiene cuatro cifras
    if (numero < 1000 || numero > 9999) {
        printf("El numero debe tener cuatro cifras.\n");
        return 1;  // Salir del programa con código de error
    }

    // Verificar si el número es primo
    if (!esPrimo(numero)) {
        printf("El numero no es primo.\n");
        return 1;  // Salir del programa con código de error
    }

    // Calcular los dígitos
    int tercer_digito = numero % 10;
    int segundo_digito = (numero / 10) % 10;

    // Verificar si la suma del segundo y tercer dígito es 10
    if (segundo_digito + tercer_digito == 10) {
        printf("El numero es primo y la suma del segundo y tercer digito es 10.\n");
    } else {
        printf("El numero es primo, pero la suma del segundo y tercer digito no es 10.\n");
    }

    return 0;  // Salir del programa sin errores
}
